static inline int
____atomic_test_and_set_bit(unsigned int bit, volatile unsigned long *p)
{
        unsigned long flags;
        unsigned int res;
        unsigned long mask = BIT_MASK(bit);

        p += BIT_WORD(bit);

        raw_local_irq_save(flags);
        res = *p;
        *p = res | mask;
        raw_local_irq_restore(flags);

        return (res & mask) != 0;
}
