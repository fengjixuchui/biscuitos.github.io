static inline void ____atomic_clear_bit(unsigned int bit, volatile unsigned long *p)
{
        unsigned long flags;
        unsigned long mask = BIT_MASK(bit);

        p += BIT_WORD(bit);

        raw_local_irq_save(flags);
        *p &= ~mask;
        raw_local_irq_restore(flags);
}
