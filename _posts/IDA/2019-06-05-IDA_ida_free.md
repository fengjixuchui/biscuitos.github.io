/**
 * ida_free() - Release an allocated ID.
 * @ida: IDA handle.
 * @id: Previously allocated ID.
 *
 * Context: Any context.
 */
void ida_free(struct ida *ida, unsigned int id)
{
        unsigned long flags;

        BUG_ON((int)id < 0);
        xa_lock_irqsave(&ida->ida_rt, flags);
        ida_remove(ida, id);
        xa_unlock_irqrestore(&ida->ida_rt, flags);
}
EXPORT_SYMBOL(ida_free);
